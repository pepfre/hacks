local dG = function(dH, dv, dw, aO, dI, i)	
	if dH == 0 then
		return dv
	end
	dH = dH / aO
	if dH == 1 then
		return dv + dw
	end
	if not i then
		i = aO * 0.3
	end
	local h = nil
	if not dI or dI < math.abs(dw) then
		dI = dw
		h = i / 4
	else
		h = i / (2 * math.pi) * math.asin(dw / dI)
	end
	return dI * math.pow(2, -10 * (dH)) * math.sin((dH * aO - h) * 2 * math.pi / (i)) + dw + dv
end
